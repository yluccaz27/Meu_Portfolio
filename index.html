Monitor de Chamados

Este projeto monitora chamados em um sistema e alerta o usuário com notificações e som.


                let valorAnterior = 0;
                
                // Reproduz som para novos chamados
                function tocarSom() {
                    const audio = new Audio(chrome.runtime.getURL("alarme.mp3"));
                    audio.play();
                }
                
                // Exibe notificação
                function mostrarNotificacao() {
                    if (Notification.permission === "granted") {
                        new Notification("Novo chamado no sistema!");
                        tocarSom();
                    }
                }
                
                // Observa mudanças no elemento #t2
                function monitorarChamado() {
                    const elemento = document.querySelector("#t2");
                    if (elemento) {
                        const observer = new MutationObserver((mutations) => {
                            mutations.forEach((mutation) => {
                                if (mutation.type === "childList") {
                                    const valorAtual = parseInt(elemento.innerText) || 0;
                                    if (valorAtual > valorAnterior) {
                                        mostrarNotificacao();
                                        valorAnterior = valorAtual;
                                    }
                                }
                            });
                        });
                        observer.observe(elemento, { childList: true, subtree: true });
                    }
                }
                monitorarChamado();
